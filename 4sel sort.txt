Selection sort 
#include <stdio.h>

void selectionSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int minIdx = i;
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[minIdx])
                minIdx = j;
        }
        int temp = arr[i];
        arr[i] = arr[minIdx];
        arr[minIdx] = temp;
    }
}

int main() {
    int n;
    printf("Enter number of products: ");
    scanf("%d", &n);

    int prices[n];
    printf("Enter product prices:\n");
    for (int i = 0; i < n; i++)
        scanf("%d", &prices[i]);

    selectionSort(prices, n);

    printf("Sorted prices (selection sort):\n");
    for (int i = 0; i < n; i++)
        printf("%d ", prices[i]);
    printf("\n");

    return 0;
}
