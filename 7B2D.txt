#include <stdio.h>
#define MAX 100
int stack[MAX];
int top = -1;
void push(int value) {
    if (top == MAX - 1) {
        printf("Stack Overflow\n");
        return;
    }
    stack[++top] = value;
}
int pop() {
    if (top == -1) {
        printf("Stack Underflow\n");
        return -1;
    }
    return stack[top--];
}
void decimalToBinary(int num) {
    if (num == 0) {
        printf("Binary equivalent: 0\n");
        return;
    }
    while (num != 0) {
        push(num % 2);
        num = num / 2;
    }
    printf("Binary equivalent: ");
    while (top != -1) {
        printf("%d", pop());
    }
    printf("\n");
}

int main() {
    int decimalNumber;
    printf("Enter a decimal number: ");
    scanf("%d", &decimalNumber);

    decimalToBinary(decimalNumber);

    return 0;
}
