#include <stdio.h>

void bubbleSort(int arr[], int n) {
    int i, j, temp;
    for(i = 0; i < n - 1; i++) {
        for(j = 0; j < n - 1 - i; j++) {
            if(arr[j] > arr[j + 1]) {
                // Swap arr[j] and arr[j+1]
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main() {
    int n, i;
    printf("Enter number of products: ");
    scanf("%d", &n);

    int prices[n];
    printf("Enter product prices:\n");
    for(i = 0; i < n; i++) 
        scanf("%d", &prices[i]);

    bubbleSort(prices, n);

    printf("Sorted prices (bubble sort):\n");
    for(i = 0; i < n; i++)
        printf("%d ", prices[i]);
    printf("\n");

    return 0;
}
