#include <stdio.h>
int main() {
    int i, j, rows;
    printf("Input number of rows: ");
    scanf("%d", &rows);
    for(i = 1; i <= rows; i++) {
        for(j = 1; j <= i; j++) {
            printf("%d", j);
        }
        printf("\n");
    }
    return 0;
}



#include <stdio.h>
int main() {
    int i, j, k, n;
    printf("Input number of rows (middlemost row): ");
    scanf("%d", &n);
    // Upper part including middle
    for(i = 1; i <= n; i++) {
        for(j = i; j < n; j++)
            printf(" ");
        for(j = 1; j <= i; j++)
            printf("%d", j);
        for(j = i - 1; j >= 1; j--)
            printf("%d", j);
        printf("\n");
    }
    // Lower part
    for(i = n - 1; i >= 1; i--) {
        for(j = n; j > i; j--)
            printf(" ");
        for(j = 1; j <= i; j++)
            printf("%d", j);
        for(j = i - 1; j >= 1; j--)
            printf("%d", j);
        printf("\n");
    }
    return 0;
}
