#include <stdio.h>

void insertionSort(int arr[], int n) {
    int i, j, key;
    for(i = 1; i < n; i++) {
        key = arr[i];
        j = i - 1;
        while(j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
}

int main() {
    int n, i;
    printf("Enter number of products: ");
    scanf("%d", &n);

    int prices[n];
    printf("Enter product prices:\n");
    for(i = 0; i < n; i++)
        scanf("%d", &prices[i]);

    insertionSort(prices, n);

    printf("Sorted prices (insertion sort):\n");
    for(i = 0; i < n; i++)
        printf("%d ", prices[i]);
    printf("\n");

    return 0;
}
