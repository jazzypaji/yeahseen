#include <stdio.h>
int main() {
    int i, j, space, rows;
    printf("Input number of rows: ");
    scanf("%d", &rows);
    for(i = 1; i <= rows; i++) {
        for(space = 1; space <= rows - i; space++)
            printf(" ");
        for(j = 1; j <= 2 * i - 1; j++)
            printf("*");
        printf("\n");
    }
    return 0;
}





#include <stdio.h>
int main() {
    int i, j, rows;
    char ch;
    printf("Input number of rows: ");
    scanf("%d", &rows);
    for(i = 1; i <= rows; i++) {
        // print spaces
        for(j = 1; j <= rows - i; j++)
            printf(" ");
        // print ascending letters
        for(ch = 'A'; ch < 'A' + i; ch++)
            printf("%c", ch);
        // print descending letters (skip last of ascending)
        for(ch = 'A' + i - 2; ch >= 'A'; ch--)
            printf("%c", ch);
        printf("\n");
    }
    return 0;
}
